  GNU nano 7.2                                           tracker.py                                                     import requests #un tool extern care ii permite python-ului sa acceseze orice de pe internet
def get_flights(): #creez o procedura numita get_flights
  url = "https://opensky-network.org/api/states/all?lamin=43.6&lomin=20.2&lamax=48.2&lomax=29.7"
  try:
    response = requests.get(url) #ne trimite direct la adresa link ului
    data=response.json() #json makes the data visible
    flights = data.get('states', []) #la states sunt listate avioanele, daca ceva lipseste, punem [] ca sa n avem erori
    print(f"sunt {len(flights)} avoane deasupra Romaniei în acest moment:\n")
    print(f"{'indicativ':<10} | {'tara origine':<15} | {'altitudine (m)':<10} | {'viteza (m/s)':<12}")
    print("-" * 65)
    for f in flights[:30]: #aratam doar primele 10 zboruri
      callsign = f[1].strip() or "N/A"
      country = f[2]
      altitude = f[7] if f[7] else "la sol"
      velocity = f[9] if f[9] is not None  else 0
      print(f"{callsign:<10} | {country:<15} | {altitude:<10} | {velocity:<12} ") #f vine de la f-string si printeaza c>  except Exception as e:
     print(f"eroare: {e}")
if __name__ == "__main__":
  get_flights()
